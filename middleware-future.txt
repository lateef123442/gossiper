import { NextResponse, type NextRequest } from 'next/server'
import { createServerClient, type CookieOptions } from '@supabase/ssr'
import { SecurityHeaders } from '@/lib/auth-security'

export const runtime = 'edge' // âœ… Enable Edge Runtime

export async function middleware(request: NextRequest) {
  // ðŸš¦ Bypass AssemblyAI webhook â€” must not require auth
  if (request.nextUrl.pathname.startsWith('/api/transcription/callback')) {
    console.log('ðŸª¶ [MIDDLEWARE] Bypassing auth for AssemblyAI webhook')
    return NextResponse.next()
  }

  // âš™ï¸ Initialize single NextResponse (Edge-safe)
  const response = NextResponse.next()

  // ðŸ§© Setup Supabase Edge client with cookie handling
  const supabase = createServerClient(
    process.env.NEXT_PUBLIC_SUPABASE_URL!,
    process.env.NEXT_PUBLIC_SUPABASE_ANON_KEY!,
    {
      cookies: {
        get: (name: string) => request.cookies.get(name)?.value,
        set: (name: string, value: string, options: CookieOptions) => {
          response.cookies.set({ name, value, ...options })
        },
        remove: (name: string, options: CookieOptions) => {
          response.cookies.set({ name, value: '', ...options })
        },
      },
    }
  )

  // ðŸ§  Get current authenticated user (lightweight)
  const { data, error } = await supabase.auth.getUser()
  const user = data?.user ?? null
  const path = request.nextUrl.pathname

  // ðŸ§± Public routes (no auth required)
  const publicRoutes = [
    '/',
    '/login',
    '/signup',
    '/forgot-password',
    '/auth',
    '/auth/callback',
    '/features',
    '/pricing',
    '/about',
    '/help',
    '/contact',
    '/privacy',
    '/terms',
  ]
  const isPublic = publicRoutes.some(route => path.startsWith(route))

  // ðŸš§ Protect private routes
  if (!user && !isPublic && !path.startsWith('/api')) {
    console.log(`ðŸ›¡ï¸ Redirect unauthenticated user from: ${path}`)
    const url = request.nextUrl.clone()
    url.pathname = '/login'
    url.searchParams.set('redirect', path)
    return NextResponse.redirect(url)
  }

  // ðŸš« Prevent logged-in users from hitting login/signup pages
  if (
    user &&
    request.method === 'GET' &&
    (path.startsWith('/login') || path.startsWith('/signup'))
  ) {
    const role = (user.user_metadata?.role as 'student' | 'lecturer' | 'admin') || 'student'
    const redirect =
      role === 'lecturer'
        ? '/dashboard?role=lecturer'
        : role === 'admin'
        ? '/dashboard?role=admin'
        : '/dashboard?role=student'

    console.log(`ðŸ” Redirecting ${user.id.slice(0, 6)} to dashboard`)
    return NextResponse.redirect(new URL(redirect, request.url))
  }

  // ðŸ‘® Role-based access control
  if (user && path.startsWith('/create-session')) {
    const role = user.user_metadata?.role || 'student'
    if (role !== 'lecturer' && role !== 'admin') {
      console.warn(`ðŸš« Unauthorized create-session by ${user.id.slice(0, 6)}`)
      return NextResponse.redirect(new URL('/dashboard', request.url))
    }
  }

  // ðŸ§¾ Attach user context headers for API calls
  if (user && path.startsWith('/api')) {
    response.headers.set('x-user-id', user.id)
    response.headers.set('x-user-role', user.user_metadata?.role || 'student')
  }

  // ðŸ›¡ï¸ Apply security headers last
  Object.entries(SecurityHeaders.getSecurityHeaders()).forEach(([key, val]) =>
    response.headers.set(key, val)
  )

  return response
}

export const config = {
  matcher: [
    '/((?!_next/static|_next/image|favicon.ico|.*\\.(?:svg|png|jpg|jpeg|gif|webp)$).*)',
  ],
}
